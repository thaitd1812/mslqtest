% ===================== II. ĐÁNH GIÁ =====================
\sect{II}{ĐÁNH GIÁ}
\subh{2.1}{Tổng quan}

\renewcommand{\arraystretch}{1.2}
\begin{tabularx}{\linewidth}{|>{\RaggedRight}p{3.5cm}|X|}
\hline
\rowcolor{brand!10}
\multicolumn{1}{|c|}{\thx{Nội dung}} & \multicolumn{1}{c|}{\thx{Chi tiết}} \\ \hline
\textbf{Điểm tốt} & \begin{itemize}
  \item Chiến lược học tập: Con biết sử dụng nhiều phương pháp học như tóm tắt, liên hệ kiến thức, ghi lỗi sai, tìm cách giải khác.
\end{itemize} \\ \hline
\textbf{Điểm cần cải thiện} & \begin{itemize}
  \item Giá trị và hứng thú học tập (cần hỗ trợ): Con chưa thật sự thấy môn học hấp dẫn hoặc chưa rõ học để làm gì.
  \item Lo âu khi kiểm tra (Áp lực cao - cần lưu ý với phụ huynh và giáo viên): Con dễ căng thẳng, lo sợ điểm kém hoặc sợ bị mắng khi làm bài kiểm tra.
\end{itemize} \\ \hline
\textbf{Nhận xét chung} & Con có nền tảng tốt ở nhóm chiến lược học tập. Nhóm cần được quan tâm nhiều hơn là giá trị và hứng thú học tập. Đặc biệt, mức lo âu khi kiểm tra đang ở ngưỡng áp lực cao - đây là yếu tố tâm lý cần được ưu tiên tháo gỡ vì có thể khiến con làm bài dưới khả năng thực tế. \\ \hline
\end{tabularx}
\vspace{8pt}

% ===================== III. ĐỀ XUẤT =====================
\sect{III}{ĐỀ XUẤT PHƯƠNG ÁN CẢI THIỆN}
\subh{3.1}{Lộ trình học}

\renewcommand{\arraystretch}{1.2}
\begin{tabularx}{\linewidth}{|>{\RaggedRight}p{3.3cm}|X|}
\hline
\rowcolor{brand!10}
\multicolumn{1}{|c|}{\thx{Giai đoạn}} & \multicolumn{1}{c|}{\thx{Nội dung}} \\ \hline
\textbf{Giai đoạn 1 (Tuần 1-4)} & Mục tiêu: Ổn định tâm lý phòng thi: cho con làm các bài kiểm tra ngắn, áp lực thấp, tăng dần độ khó để con quen với cảm giác thi cử và bớt căng thẳng. \\ \hline
\textbf{Giai đoạn 2 (Tuần 5-8)} & Mục tiêu: Trang bị phương pháp học hiệu quả cho nhóm giá trị và hứng thú học tập. Hướng dẫn con kỹ năng tóm tắt bài, vẽ sơ đồ, ghi chú lỗi sai và tự đặt câu hỏi kiểm tra mức độ hiểu. \\ \hline
\textbf{Giai đoạn 3 (Tuần 9-12)} & Mục tiêu: Rèn cho con khả năng tự lập kế hoạch, tự đặt mục tiêu và tự đánh giá tiến độ học. Nâng dần các nhóm đang ở mức Khá lên mức Tốt và duy trì thói quen học tập bền vững. \\ \hline
\end{tabularx}
\vspace{7pt}

\subh{3.2}{Khuyến nghị cho phụ huynh}
Phụ huynh nên đồng hành với con bằng cách:
\begin{itemize}
  \item Tránh tạo thêm áp lực điểm số; ghi nhận và động viên nỗ lực của con thay vì chỉ nhìn vào kết quả bài thi.
  \item Kết nối môn Toán với sở thích và đời sống thực tế để con cảm nhận được ý nghĩa của việc học.
\end{itemize}
